    data() {
        return {
            type: this.$route.params.type,
            items: [],
        }
    },
    watch: {
        '$route': 'fetchItems'
    },
    created() {
        this.fetchItems(0)
    },
    methods: {
        fetchItems() {
            this.type = this.$route.params.type
            this.item = []
            let initial_ids = [1, 13, 14]

            for (let i in initial_ids) {
                
                let id = initial_ids[i]
                console.log('id', id)
                fetch(`http://swapi.dev/api/${this.type}/${id}`, {

                })

            }
        }
    }
}
